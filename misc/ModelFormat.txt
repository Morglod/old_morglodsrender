Сокращения:
uint - unsigned int
uchar - unsigned char
loop n - повторяющиеся данные, количеством n

-------------------------------------------------------------------
uint кол-во мешей
unit кол-во материалов
float[16] матрица трансформаций рут нода

loop кол-во мешей {
	uint длина имени меша
	char* имя меша (если длина != 0)
	float[16] матрица трансформаций
	uint id материала
	uint кол-во вершин
	int opengl stride
	int кол-во типов данных в вершине (позиция, текстурные координаты, нормаль, цвет)

	loop кол-во типов данных в вершине {
		uchar флаг типа данных ( 0 - позиция, 1 - текстурн. коорд, 2 - нормаль, 3 - цвет)
	}
	
	uint размер буфера вершин в байтах (NumVertices * gl_stride * sizeof(float))
	float[] буфер вершин {
		данные повторяются для каждой вершине в таком порядке (если существуют):
		float[3] позиция
		float[2] текстурные координаты
		float[3] нормаль
		float[4] цвет rgba
	}
	
	uint кол-во индексов/3 (фейсов)
	uint размер буфера индексов в байтах (NumFaces * 3 * sizeof(unsigned int))
	uint[] буфер индексов
}
